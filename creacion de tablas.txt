-- Crear base de datos SAAS_PEOPLE si no existe
CREATE DATABASE IF NOT EXISTS SAAS_PEOPLE;
USE SAAS_PEOPLE;

-- Create table PEOPLE if it does not exist
CREATE TABLE IF NOT EXISTS PEOPLE(
    ID INT NOT NULL PRIMARY KEY,
    NAME VARCHAR(50) NOT NULL,
    LASTNAME VARCHAR(50) NOT NULL,
    PHONE INT NOT NULL,
    AGE INT NOT NULL,
    EMAIL VARCHAR(255) NOT NULL  -- Added email column
);
